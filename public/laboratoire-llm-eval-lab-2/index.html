Étape 1 · Créer le module de métriques

Dans le Terminal, collez ce bloc. Il crée le dossier metrics/ et écrit evaluator.py, qui contient les trois mesures et le calcul du score global. Le module est ensuite rendu exécutable. 
mkdir -p /root/evaluation/metrics

cat > /root/evaluation/metrics/evaluator.py << 'EOF'
#!/usr/bin/env python3
"""Evaluation metrics for LLM responses."""
import numpy as np
from typing import List, Dict, Any
from sentence_transformers import SentenceTransformer


class EvaluationMetrics:
    """Three complementary metrics plus a weighted overall score."""

    def __init__(self, embedding_model: str = "all-MiniLM-L6-v2"):
        self.model = SentenceTransformer(embedding_model)

    def keyword_match(self, generated: str, keywords: List[str]) -> Dict[str, Any]:
        text = generated.lower()
        matched = [kw for kw in keywords if kw.lower() in text]
        missing = [kw for kw in keywords if kw.lower() not in text]
        score = len(matched) / len(keywords) if keywords else 0.0
        return {"score": score, "matched": matched, "missing": missing}

    def semantic_similarity(self, generated: str, reference: str) -> Dict[str, Any]:
        gen_emb, ref_emb = self.model.encode([generated, reference])
        cosine = float(np.dot(gen_emb, ref_emb) /
                       (np.linalg.norm(gen_emb) * np.linalg.norm(ref_emb)))
        return {"score": cosine, "label": self._label(cosine)}

    def _label(self, score: float) -> str:
        if score >= 0.8:
            return "excellent"
        if score >= 0.6:
            return "good"
        if score >= 0.4:
            return "fair"
        return "poor"

    def length_ratio(self, generated: str, reference: str) -> Dict[str, Any]:
        gen_len = len(generated.split())
        ref_len = len(reference.split())
        ratio = gen_len / ref_len if ref_len else 0.0
        if ratio > 1.5:
            note = "verbose"
        elif ratio < 0.5:
            note = "brief"
        else:
            note = "appropriate"
        return {"ratio": ratio, "note": note}

    def evaluate(self, generated: str, reference: str,
                 keywords: List[str]) -> Dict[str, Any]:
        return {
            "keyword_match": self.keyword_match(generated, keywords),
            "semantic_similarity": self.semantic_similarity(generated, reference),
            "length_ratio": self.length_ratio(generated, reference),
        }


def overall_score(result: Dict[str, Any]) -> float:
    weights = {"keyword_match": 0.3, "semantic_similarity": 0.5, "length_ratio": 0.2}
    score = result["keyword_match"]["score"] * weights["keyword_match"]
    score += result["semantic_similarity"]["score"] * weights["semantic_similarity"]
    ratio = result["length_ratio"]["ratio"]
    length_score = max(0.0, 1 - abs(1 - ratio) * 0.5)
    score += length_score * weights["length_ratio"]
    return score


if __name__ == "__main__":
    metrics = EvaluationMetrics()
    generated = "View pod logs with kubectl logs. Use kubectl logs my-pod and add -f to stream."
    reference = "Use the kubectl logs command followed by the pod name, for example kubectl logs my-pod."
    keywords = ["kubectl", "logs"]
    result = metrics.evaluate(generated, reference, keywords)
    print("Evaluation result")
    print("  keyword_match       : %.2f" % result["keyword_match"]["score"])
    print("  semantic_similarity : %.2f (%s)" % (result["semantic_similarity"]["score"],
                                                  result["semantic_similarity"]["label"]))
    print("  length_ratio        : %.2f (%s)" % (result["length_ratio"]["ratio"],
                                                  result["length_ratio"]["note"]))
    print("  overall_score       : %.2f" % overall_score(result))
EOF

Étape 3 · Écrire et lancer les tests

Créez un petit script de test qui passe trois réponses types (bonne, partielle, trop longue) dans le module et affiche les scores. Collez ce bloc, puis exécutez-le. 
cat > /root/evaluation/metrics/test_metrics.py << 'EOF'
#!/usr/bin/env python3
"""Run the metrics on a few sample answers."""
from evaluator import EvaluationMetrics, overall_score

metrics = EvaluationMetrics()

cases = [
    {
        "name": "Good answer",
        "generated": "Use kubectl logs my-pod to read a pod's logs. Add -f to stream them.",
        "reference": "Use the kubectl logs command followed by the pod name, for example kubectl logs my-pod.",
        "keywords": ["kubectl", "logs"],
    },
    {
        "name": "Partial answer",
        "generated": "You can check the logs from the terminal.",
        "reference": "Use the kubectl logs command followed by the pod name.",
        "keywords": ["kubectl", "logs"],
    },
    {
        "name": "Verbose answer",
        "generated": "To read the logs of a Kubernetes pod you use the kubectl command line tool. "
                     "The command is kubectl logs followed by the pod name. You can add -f to "
                     "stream in real time and -c to target a container in a multi-container pod.",
        "reference": "Use the kubectl logs command followed by the pod name.",
        "keywords": ["kubectl", "logs"],
    },
]

print("Metrics test report")
print("=" * 40)
for case in cases:
    result = metrics.evaluate(case["generated"], case["reference"], case["keywords"])
    print("\n" + case["name"])
    print("  keyword_match       : %.2f" % result["keyword_match"]["score"])
    print("  semantic_similarity : %.2f (%s)" % (result["semantic_similarity"]["score"],
                                                  result["semantic_similarity"]["label"]))
    print("  length_ratio        : %.2f (%s)" % (result["length_ratio"]["ratio"],
                                                  result["length_ratio"]["note"]))
    print("  overall_score       : %.2f" % overall_score(result))
EOF
