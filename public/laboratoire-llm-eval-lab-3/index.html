Étape 1 · Créer le pipeline et l'exécuter

Collez ce bloc dans le Terminal. Il écrit pipeline.py dans metrics/ : la classe orchestre le chargement du dataset, l'évaluation de chaque réponse, l'agrégation et l'export. Le bloc lance ensuite un premier passage. 
cat > /root/evaluation/metrics/pipeline.py << 'EOF'
#!/usr/bin/env python3
"""Complete evaluation pipeline."""
import json
import sys
from typing import Dict, List, Any, Optional
from pathlib import Path
from datetime import datetime

from evaluator import EvaluationMetrics, overall_score


class EvaluationPipeline:
    """Loads a dataset, scores answers and aggregates the results."""

    def __init__(self, dataset_path: Optional[str] = None):
        self.dataset_path = dataset_path or "/root/evaluation/datasets/kubernetes_eval.json"
        self.metrics = EvaluationMetrics()
        self.dataset = []
        self.results = []
        self.summary = {}

    def load_dataset(self) -> bool:
        try:
            with open(self.dataset_path, encoding="utf-8") as fh:
                self.dataset = json.load(fh)
            print("Dataset loaded: %d entries" % len(self.dataset))
            return True
        except Exception as err:
            print("Error loading dataset: %s" % err)
            return False

    def run(self, answers: Dict[str, str]) -> Dict[str, Any]:
        self.results = []
        summary = {"total": len(self.dataset), "evaluated": 0, "scores": [],
                   "by_category": {}, "by_difficulty": {},
                   "timestamp": datetime.now().isoformat()}
        for item in self.dataset:
            if item["id"] not in answers:
                continue
            res = self.metrics.evaluate(answers[item["id"]],
                                        item["reference_answer"],
                                        item["expected_keywords"])
            score = overall_score(res)
            self.results.append({
                "id": item["id"], "question": item["question"],
                "category": item.get("category", "unknown"),
                "difficulty": item.get("difficulty", "unknown"),
                "generated": answers[item["id"]],
                "keyword_match": res["keyword_match"],
                "semantic_similarity": res["semantic_similarity"],
                "length_ratio": res["length_ratio"],
                "overall_score": score,
            })
            summary["evaluated"] += 1
            summary["scores"].append(score)
            cat = item.get("category", "unknown")
            dif = item.get("difficulty", "unknown")
            summary["by_category"].setdefault(cat, {"scores": [], "count": 0})
            summary["by_category"][cat]["scores"].append(score)
            summary["by_category"][cat]["count"] += 1
            summary["by_difficulty"].setdefault(dif, {"scores": [], "count": 0})
            summary["by_difficulty"][dif]["scores"].append(score)
            summary["by_difficulty"][dif]["count"] += 1
        if summary["scores"]:
            summary["average_score"] = sum(summary["scores"]) / len(summary["scores"])
            for group in ("by_category", "by_difficulty"):
                for key in summary[group]:
                    s = summary[group][key]["scores"]
                    summary[group][key]["average"] = sum(s) / len(s)
        self.summary = summary
        return summary

    def print_summary(self):
        if not self.summary:
            print("No summary available")
            return
        print("\n" + "=" * 60)
        print("EVALUATION SUMMARY")
        print("=" * 60)
        print("Evaluated   : %d / %d" % (self.summary["evaluated"], self.summary["total"]))
        print("Average     : %.3f" % self.summary.get("average_score", 0))
        print("\nBy category")
        for cat, data in self.summary.get("by_category", {}).items():
            print("  %-14s %d entries  avg %.3f" % (cat, data["count"], data["average"]))
        print("\nBy difficulty")
        for dif, data in self.summary.get("by_difficulty", {}).items():
            print("  %-14s %d entries  avg %.3f" % (dif, data["count"], data["average"]))

    def save_results(self, path: str):
        Path(path).parent.mkdir(parents=True, exist_ok=True)
        with open(path, "w", encoding="utf-8") as fh:
            json.dump({"summary": self.summary, "results": self.results}, fh, indent=2)
        print("Results saved to %s" % path)

    def export_report(self, path: str):
        lines = ["=" * 60, "EVALUATION REPORT", "=" * 60,
                 "Generated   : %s" % datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
                 "Dataset     : %s" % self.dataset_path,
                 "Average     : %.3f" % self.summary.get("average_score", 0), ""]
        for res in self.results:
            lines.append("%s  score %.3f  (%s)" % (res["id"], res["overall_score"], res["category"]))
        Path(path).parent.mkdir(parents=True, exist_ok=True)
        with open(path, "w", encoding="utf-8") as fh:
            fh.write("\n".join(lines))
        print("Report saved to %s" % path)


def sample_answers() -> Dict[str, str]:
    return {
        "k8s-eval-01": "Use kubectl logs my-pod -f to stream a pod's logs in real time.",
        "k8s-eval-02": "A pod stays Pending when the scheduler cannot place it, often due to insufficient memory or taints on the nodes.",
        "k8s-eval-03": "Scale it with kubectl scale deployment myapp --replicas=4.",
        "k8s-eval-04": "Use a StatefulSet for stateful apps needing persistent storage and stable identities; a Deployment suits stateless replicas.",
        "k8s-eval-05": "Create it with kubectl create secret generic db --from-literal=password=secret.",
        "k8s-eval-06": "Roll back with kubectl rollout undo deployment/myapp to restore the previous revision.",
    }


if __name__ == "__main__":
    pipeline = EvaluationPipeline()
    if not pipeline.load_dataset():
        sys.exit(1)
    pipeline.run(sample_answers())
    pipeline.print_summary()
    pipeline.save_results("/root/evaluation/results/pipeline_results.json")
    pipeline.export_report("/root/evaluation/results/pipeline_report.txt")
    print("\nPipeline run complete.")
EOF

Étape 2 · Générer le rapport final

Le générateur réutilise le pipeline et écrit un rapport JSON exploitable (scores par catégorie et difficulté). Collez ce bloc, puis exécutez-le. 
cat > /root/evaluation/generate_report.py << 'EOF'
#!/usr/bin/env python3
"""Produce the final evaluation report (text + JSON)."""
import json
import sys
from pathlib import Path
from datetime import datetime

sys.path.append("/root/evaluation/metrics")
from pipeline import EvaluationPipeline, sample_answers


def generate_report():
    print("GENERATING EVALUATION REPORT")
    print("=" * 60)
    Path("/root/evaluation/results").mkdir(parents=True, exist_ok=True)

    pipeline = EvaluationPipeline()
    if not pipeline.load_dataset():
        print("ERROR: failed to load dataset")
        return False

    pipeline.run(sample_answers())
    pipeline.print_summary()
    pipeline.save_results("/root/evaluation/results/evaluation_results.json")
    pipeline.export_report("/root/evaluation/results/evaluation_report.txt")

    report = {
        "timestamp": datetime.now().isoformat(),
        "dataset_path": pipeline.dataset_path,
        "statistics": {
            "total_questions": len(pipeline.results),
            "average_score": pipeline.summary.get("average_score", 0),
            "by_category": pipeline.summary.get("by_category", {}),
            "by_difficulty": pipeline.summary.get("by_difficulty", {}),
        },
        "results": pipeline.results,
    }
    out = "/root/evaluation/results/evaluation_report.json"
    with open(out, "w", encoding="utf-8") as fh:
        json.dump(report, fh, indent=2)

    print("\nJSON report saved to %s" % out)
    print("  total_questions : %d" % report["statistics"]["total_questions"])
    print("  average_score   : %.3f" % report["statistics"]["average_score"])
    print("  categories      : %s" % ", ".join(report["statistics"]["by_category"].keys()))
    return True


if __name__ == "__main__":
    sys.exit(0 if generate_report() else 1)
EOF
